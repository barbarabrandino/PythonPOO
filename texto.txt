# Superclasse
class Funcionario:
    def __init__(self, nome):
        self.nome = nome

    def calcular_salario(self):
        raise NotImplementedError("Este método deve ser implementado nas subclasses.")


# Subclasse CLT
class FuncionarioCLT(Funcionario):
    def __init__(self, nome, salario_mensal):
        super().__init__(nome)
        self.salario_mensal = salario_mensal

    def calcular_salario(self):
        # CLT recebe salário fixo
        return self.salario_mensal


# Subclasse PJ
class FuncionarioPJ(Funcionario):
    def __init__(self, nome, valor_hora, horas_trabalhadas):
        super().__init__(nome)
        self.valor_hora = valor_hora
        self.horas_trabalhadas = horas_trabalhadas

    def calcular_salario(self):
        # PJ é pago por hora
        return self.valor_hora * self.horas_trabalhadas


# Exemplo de uso
clt = FuncionarioCLT("João", 3000)
pj = FuncionarioPJ("Maria", 100, 80)

print(f"{clt.nome} (CLT) - Salário: R${clt.calcular_salario():.2f}")
print(f"{pj.nome} (PJ) - Salário: R${pj.calcular_salario():.2f}")


# Superclasse
class Mensagem:
    def __init__(self, destinatario, conteudo):
        self.destinatario = destinatario
        self.conteudo = conteudo

    def enviar_mensagem(self):
        raise NotImplementedError("Este método deve ser implementado pelas subclasses.")


# Subclasse: Email
class Email(Mensagem):
    def enviar_mensagem(self):
        print(f"Enviando Email para {self.destinatario}: {self.conteudo}")


# Subclasse: SMS
class SMS(Mensagem):
    def enviar_mensagem(self):
        print(f"Enviando SMS para {self.destinatario}: {self.conteudo}")


# Subclasse: Whatsapp
class Whatsapp(Mensagem):
    def enviar_mensagem(self):
        print(f"Enviando mensagem no WhatsApp para {self.destinatario}: {self.conteudo}")


# Exemplo de uso
email = Email("joao@email.com", "Olá João, tudo bem?")
sms = SMS("+559199999999", "Sua verificação é 123456.")
whatsapp = Whatsapp("+559198888888", "Oi! Chegarei às 18h.")

email.enviar_mensagem()
sms.enviar_mensagem()
whatsapp.enviar_mensagem()
